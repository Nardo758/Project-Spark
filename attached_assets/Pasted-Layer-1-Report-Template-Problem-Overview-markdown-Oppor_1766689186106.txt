Census Data Query for Layer 2 Market Sizing
sql-- Comprehensive market sizing query
WITH target_demographics AS (
  SELECT 
    g.name as geography_name,
    g.geography_type,
    SUM(d.B01003_001E) as total_population,
    SUM(d.B11001_001E) as total_households,
    AVG(d.B19013_001E) as median_income,
    AVG(d.B01002_001E) as median_age,
    SUM(d.B15003_022E + d.B15003_023E + d.B15003_024E + d.B15003_025E) as educated_population,
    AVG(d.B25077_001E) as median_home_value,
    AVG(d.B25064_001E) as median_rent,
    SUM(d.B23025_005E)::FLOAT / NULLIF(SUM(d.B23025_002E), 0) as unemployment_rate
  FROM census_acs_5year d
  JOIN census_geographies g ON d.geography_id = g.geography_id
  WHERE ST_DWithin(
    g.geography_boundary::geography,
    (SELECT coverage_area FROM opportunities WHERE id = '[opportunity_id]'),
    0  -- Within coverage area
  )
  AND d.year = 2023
  GROUP BY g.name, g.geography_type
),
market_segments AS (
  SELECT 
    geography_name,
    total_households,
    median_income,
    -- Apply filters based on opportunity target demographic
    CASE 
      WHEN median_age BETWEEN 25 AND 45 
        AND median_income > 75000
        AND educated_population::FLOAT / total_population > 0.3
      THEN total_households * 0.4  -- 40% fit criteria
      ELSE total_households * 0.15  -- 15% fit criteria
    END as target_households,
    median_income * total_households as purchasing_power
  FROM target_demographics
)
SELECT 
  SUM(target_households) as total_target_households,
  SUM(purchasing_power) as total_purchasing_power,
  AVG(median_income) as avg_target_income,
  -- TAM calculation
  SUM(target_households * median_income * 0.02) as estimated_tam,  -- 2% of income
  -- SAM calculation (top 50 metros)
  SUM(target_households * median_income * 0.02) * 0.65 as estimated_sam,  -- 65% coverage
  -- SOM calculation (first year)
  SUM(target_households * median_income * 0.02) * 0.65 * 0.01 as estimated_som  -- 1% market penetration
FROM market_segments;

SQL for Geographic Analysis
sql-- Top markets by opportunity score
WITH market_metrics AS (
  SELECT 
    m.cbsa_code,
    m.cbsa_name,
    -- Demographics
    SUM(c.B01003_001E) as population,
    SUM(c.B11001_001E) as households,
    AVG(c.B19013_001E) as median_income,
    AVG(c.B01002_001E) as median_age,
    
    -- Problem density
    COUNT(DISTINCT s.id) as mention_count,
    (COUNT(DISTINCT s.id)::FLOAT / SUM(c.B01003_001E) * 100000) as problem_density,
    
    -- Competitive density
    COUNT(DISTINCT comp.id) as competitor_count,
    
    -- Economic indicators
    AVG(c.B25077_001E) as median_home_value,
    SUM(c.B23025_005E)::FLOAT / NULLIF(SUM(c.B23025_002E), 0) as unemployment_rate,
    SUM(c.B08301_019E)::FLOAT / NULLIF(SUM(c.B08301_001E), 0) as wfh_rate
    
  FROM metro_areas m
  JOIN census_acs_5year c ON c.cbsa = m.cbsa_code
  LEFT JOIN scraped_sources s ON ST_Within(s.business_location, m.boundary)
    AND s.opportunity_id = '[opportunity_id]'
  LEFT JOIN competitors comp ON ST_Within(comp.location, m.boundary)
  WHERE c.year = 2023
  GROUP BY m.cbsa_code, m.cbsa_name
),
opportunity_scores AS (
  SELECT 
    *,
    -- Calculate opportunity score (0-100)
    (
      (LEAST(problem_density, 50) / 50 * 30) +  -- Problem density (30%)
      (LEAST((households * median_income) / 1000000000, 100) * 0.25) +  -- Purchasing power (25%)
      (CASE 
        WHEN competitor_count < 3 THEN 20
        WHEN competitor_count < 6 THEN 15
        WHEN competitor_count < 10 THEN 10
        ELSE 5
      END) +  -- Competitive gap (20%)
      (CASE
        WHEN median_income > 90000 AND median_age BETWEEN 25 AND 45 THEN 15
        WHEN median_income > 75000 THEN 12
        WHEN median_income > 60000 THEN 8
        ELSE 5
      END) +  -- Demographics fit (15%)
      (CASE
        WHEN unemployment_rate < 0.04 AND wfh_rate > 0.15 THEN 10
        WHEN unemployment_rate < 0.05 THEN 7
        ELSE 5
      END)  -- Growth/modernity indicators (10%)
    ) as opportunity_score
  FROM market_metrics
)
SELECT 
  cbsa_name as metro_area,
  ROUND(opportunity_score, 0) as score,
  TO_CHAR(households, 'FM999,999,999') as target_households,
  TO_CHAR(median_income, 'FM$999,999') as median_income,
  ROUND(problem_density, 1) || '/100k' as problem_density,
  competitor_count,
  TO_CHAR(households * median_income, 'FM$999,999,999,999') as purchasing_power
FROM opportunity_scores
ORDER BY opportunity_score DESC
LIMIT 10;
