Database Extensions for Admin:
sql
-- Audit logging for admin actions
CREATE TABLE admin_audit_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  admin_id UUID REFERENCES admin_users(id),
  action VARCHAR(255) NOT NULL,
  resource_type VARCHAR(50),
  resource_id UUID,
  changes JSONB,
  ip_address INET,
  user_agent TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Admin notifications
CREATE TABLE admin_notifications (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  admin_id UUID REFERENCES admin_users(id),
  type VARCHAR(50) NOT NULL, -- 'alert', 'report', 'system', 'user'
  title VARCHAR(255) NOT NULL,
  message TEXT,
  data JSONB,
  is_read BOOLEAN DEFAULT FALSE,
  priority INTEGER DEFAULT 0, -- 0=low, 1=medium, 2=high, 3=critical
  created_at TIMESTAMP DEFAULT NOW()
);

-- Scheduled tasks and reports
CREATE TABLE admin_scheduled_tasks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  task_type VARCHAR(50) NOT NULL,
  schedule VARCHAR(50), -- 'daily', 'weekly', 'monthly', 'cron'
  last_run TIMESTAMP,
  next_run TIMESTAMP,
  config JSONB,
  status VARCHAR(50) DEFAULT 'active',
  created_at TIMESTAMP DEFAULT NOW()
);
